195  // rotation of the front two axils - turning the chassis left or right
196  if (the_app->is_key_down('A') || the_app->is_key_down(key_left)) {
197  if (axil_direction_limit > -(max_angle * 3.14159265f / 180.0f))
198  {
199  axil_direction_limit -= step_angle * 3.14159265f / 180.0f;
200  rotate_axils(axil_direction_limit);
201  }
202  }
203 
204  else if (the_app->is_key_down('D') || the_app->is_key_down(key_right)) {
205  if (axil_direction_limit < (max_angle * 3.14159265f / 180.0f)){
206  axil_direction_limit += step_angle * 3.14159265f / 180.0f;
207  rotate_axils(axil_direction_limit);
208  }
209  }
210 
211  else if (!the_app->is_key_down('A') || !the_app->is_key_down('D') || the_app->is_key_down(key_left) || the_app->is_key_down(key_right)){
212  if (axil_direction_limit != 0){
213  axil_direction_limit = 0.0f;
214  rotate_axils(axil_direction_limit);
215  }
216  }
217 
218 
219  //moving the car forwards
220  if (the_app->is_key_down('W') || the_app->is_key_down(key_up)){
221  if (motor_velocity < max_velocity){
222  motor_velocity += step_acceleration;
223  }
224  }
225 
226  //moving the car backwards - maximum velocity is less as we're moving backwards
227  else if (the_app->is_key_down('S') || the_app->is_key_down(key_down)){
228  if (motor_velocity > -max_velocity){
229  motor_velocity -= step_acceleration;
230  }
231  }
232 
233  //if no force is being applied - setting motor_velocity to 0 will produce a fake form of friction
234  else if (!the_app->is_key_down('W') || !the_app->is_key_down('S') || the_app->is_key_down(key_up) || the_app->is_key_down(key_down)){
235  if (motor_velocity != 0){
236  motor_velocity = 0.0f;
237  }
238  }
239 
240  move_direction(motor_velocity, 10); //apply the motor_velocity
241 
242  //if the xbox controller has been connected
243  if (xbox_controller.refresh()){
244  //right trigger is acceleration, left trigger is decceleration.
245  motor_velocity = xbox_controller.right_trigger - xbox_controller.left_trigger;
246  move_direction(motor_velocity, 10);
247  //turn wheels based on x pos of left analog stick
248  if (!xbox_controller.analog_deadzone()){
249  rotate_axils(xbox_controller.left_analog_x);
250  }

253  }
254  }
255  //mute sound

257  {

259  }
260 
261  sound_control();
262

265  {
266  exit(1); //exits the program....safely?
267  }

270  }
